## Abort
- abort error_code //error code is of type u64
```rust
abort 10;
assert!(expr, 10);
```

```rust
module my_addrx::Errors
{
    use std::debug::print;
    use std::string::utf8;

    fun isEven(num:u64)
    {
        if(num%2==0)
        {
            print(&utf8(b"No Error as the Number is Even"));
        }
        else
        {
            abort 11 //throwing error as the given number is not even
        }
    }

    #[test]
    fun testing()
    {
        isEven(3);
    }
}
```
