## Integer
```rust
module primary_data::Integer {
    use std::debug::print;

    fun integer_types() {
        let a: u8 = 10;
        let b: u64 = 1_000;
        let c: u128 = 10_000;
        print(&a);
        print(&b);
        print(&c);
    }

    #[test]
    fun test_integer_types() {
        integer_types()
    }
}
```
